\begin{exercice}
\
\begin{enumerate}

\item Calculer $\frac 1 2 \times \frac 2 3 \times \frac 3 4 \times \frac 4 5 \times \frac 5 6 \times \frac 6 7 \times \frac 7 8 \times \frac 8 9 \times \frac 9 {10} \times \frac {10} {11} \times \frac {11} {12}$

\item De manière plus générale, pour un entier positif $n$, calculer $\frac 1 2 \times \frac 2 3 \times \ldots \times \frac {n-1} n$.

\end{enumerate}

\end{exercice}
